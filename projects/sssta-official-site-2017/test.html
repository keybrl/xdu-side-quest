优雅的语言

以下是七种语言版本的"Hello World!程序"

请选择你觉得优雅的语言

#include <stdio.h>
int main()
{
    printf("Hello World!");
    return 0;
}

#include <iostream>
using namespace std;
int main()
{
    cout << "Hello World!";
    return 0;
}

using System;
public class hello
{
    public static void Main()
    {
        Console.WriteLine("Hello World!");
    }
}

#import <Foundation/Foundation.h>
int main (int argc, const char * argv[])
{
    NSAutoreleasePool *pool = [[NSAutoreleasePool alloc] init];
    NSLog (@"Hello, World!");
    [pool drain];
    return 0;
}

import java.io.*;
class hello
{
    public static void main (String[] args) throws java.lang.Exception
    {
        System.out.println("Hello World!");
    }
}

print 'Hello World!'

alert('Hello World!');
