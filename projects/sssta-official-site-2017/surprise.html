    int password;

        printf("Password: G-");  // |
        scanf("%d", &password);  // |
    } while (password != 302);   // |
    // ------------------------------

    BubbbleSort(key1, 7);
    for(i = 0;i < 7;i++)
        printf("%c", key1[key2[i]]);
    printf("\n");
    return 0;
}

void BubbbleSort(int *data,int n)  //冒泡排序
{
    int i, j, temp;
    for (i = n;i > 0;i--) {
        for (j = 0;j < i - 1;j++) {
            if (data[j] > data[j+1]) {
                temp = data[j];
                data[j] = data[j+1];
                data[j+1] = temp;
            }
        }
    }
}
